def convertToDocuments(articlesArray):
    try:
        raw_documents = []
        for article in articlesArray:
            #inserting documents in lower text format into raw_document list
            raw_documents.append(str(article).lower())
        return raw_documents
    except Exception as e:
        print("Error in Method convertToDocuments : ",e)

def getStopWords():
    try:
        custom_stop_words = ['news','headline','said','cent','year','time','month','people'] #custom stop word list
        custom_stop_words.extend(stop_words1) #stop word list from NLTK
        custom_stop_words.extend(stop_words2) #stop word list from SKLEARN
        final_stopwords = set(custom_stop_words) #Final set of stop_words
        return final_stopwords
    except Exception as e:
        print("Error in getStopWords ," ,e)

2. Applying Tokenization, lemmatisation, token length & Stop-word Filtering¶

In [34]:
# 
# Method Name: [preprocessCorpus]
# This method preprocesses the corpus - Applies Tokenization, Stop Word Filtering, Lemmatisation 
# @return tokens after applying preprocessing
#
def preprocessCorpus(raw_document):
    
    try:
        tokens = []
        stopWords = getStopWords() #Fetching the stop word list
        for doc in raw_document:
            doc = str(doc)
            try:
                tokensExtracted = tokenize(doc)
                for tok in tokensExtracted:
                    #Lemmatizing
                    lemmedTok = lemmatizer.lemmatize(tok)
                    #if the token is not a regular expression and is a string, its length is more than 3 and is not in the list of stopwords
                    if re.search('[a-zA-Z]', lemmedTok) and len(lemmedTok)>3 and lemmedTok not in stopWords:
                        tokens.append(lemmedTok)
            except Exception as e:
                pass
        return tokens
    except Exception as e:
        print("Some error in preprocessCorpus method ",e)

uniqToks = {}
def fetchUniqueTokens(tokens_list):
    try:
        for token in tokens_list:
            if token in uniqToks:
                uniqToks[token] += 1
            else:
                uniqToks[token] = 1
    except Exception as e:
        print("Error while fecthing the unique tokens")
    return uniqToks

Step 6 : CLUSTERING¶

This is the main excercise of this project. The aim is to cluster the documents into groups based on the similarity between them. For this particular poject I have used the example of K-Mean and Agglomerative clustering.

K-Mean is an example of partitional clustering algorithms and is based on finding the cluster centroid to partition the data into clusters whereas the Agglomerative algorithms start with each item assigned to its own cluster, and then apply a bottom-up strategy where, at each step, the most similar (least distant) pair of clusters are merged. We continue until we reach a specified number of clusters or until every item is in a single cluster. It uses distance for the evaluation. Here cosine distance is used to solve the purpose.
